  住院日报

  TO_CHAR ( 日期, 'YYYY-MM-DD HH24:MI:SS' ) >= TO_CHAR ( SYSDATE - INTERVAL '2' DAY, 'YYYY-MM-DD' ) || ' 00:00:00'
